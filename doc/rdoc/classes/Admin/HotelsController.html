def create
    @hotel = Hotel.new(hotel_params)

    respond_to do |format|
       if @hotel.save
            format.html { redirect_to admin_hotels_path }
        else
            format.html { render 'new' }
        end
    end
end

def update

        if @hotel.update_attributes(hotel_params)
            format.html { redirect_to admin_hotels_path }
        else
            format.html { render 'edit' }
        end
    end
